1 使用1行语句

print('\n'.join([' '.join(["%2s x%2s = %2s" % (j, i, i*j) for j in range(1, i+1)]) for i in range(1, 10)]))

2 使用while-while

i = 1
while i <= 9:
    j = 1
    while(j <= i):    # j的大小是由i来控制的
        print('%d*%d=%-3d' % (i, j, i*j), end='\t')
        j += 1
    print('')
    i += 1

3 使用while-for

i = 1
while (i <= 9):
    for j in range(1, i + 1):  # range()函数左闭右开
        print('%d*%d=%-3d' % (i, j, i * j), end='')
    i += 1
    print()

4 使用for-while

for i in range(1, 10):
    j = 0
    while j < i:
        j += 1
        print("%d*%d=%-3d" % (i, j, i * j), end='')
    print()

5 使用for-while 循环后加一行提示

for i in range(1, 10):  # 左闭右开
    j = 1
    while j <= i:
        print("%d*%d=%-3d" % (j, i, j*i), end='')
        j += 1
    print()
print("上边的是九九乘法口诀。")

6 使用while-for

i = 1
while (i <= 9):
    for j in range(1, i+1):  # range()函数左闭右开
        print('%d*%d=%-3d' % (i, j, i*j), end='')
    i += 1
    print()

7 使用for-for

# 九九乘法表  for-for
for i in range(1, 10):
    for j in range(1, i+1):
        print('{}x{}={}\t'.format(j, i, i*j), end='')
    print()
